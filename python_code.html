S = data[['Adj Close']]

def Volatility(S):
    re = np.log(S/S.shift(1)).dropna()
    return re.std()*np.sqrt(252)

def onetouch_put(S,E,T,t,r,sigma):
    d1 = (np.log(S/E)+(r+0.5*sigma**2)*(T-t))/(sigma*np.sqrt(T-t))
    d2 = (np.log(S/E)-(r+0.5*sigma**2)*(T-t))/(sigma*np.sqrt(T-t))
    Price = (S/E)*norm.cdf(-d1)+(S/E)**(-2*r/sigma**2)*norm.cdf(-d2)
    delta = 1/E*norm.cdf(-d1)-S/E*(1/(S*sigma*np.sqrt(T-t)))*norm.pdf(-d1)-(2*r/sigma**2)*(1/E)**(-2*r/sigma**2)*S**(-(2*r/sigma**2)-1)*norm.cdf(-d2)-(S/E)**(-2*r/sigma**2)*norm.pdf(-d2)*(1/(S*sigma*np.sqrt(T-t)))
    return Price,delta

def vanilla_digital_put(S,E,T,t,r,sigma):
    h = (np.log(S/E)+(r-0.5*sigma**2)*(T-t))/(sigma*np.sqrt(T-t))
    Price = np.exp(-r*(T-t))*norm.cdf(-h)
    return Price

def simulation(s0,K,r,sigma,T,n):
    c = 0
    def MonteCarlo():
        s = []
        s.append(s0)
        for i in range(1,int(252*T)):
            s.append(s[i-1]*np.exp((r-0.5*(sigma**2))*(1/252)+sigma*np.sqrt(1/252)*np.random.randn()))
            if s[i]<=K:
                return np.exp(-r*i/252)
        return 0
    for i in range(n):
        c+=MonteCarlo()
    return c/n
